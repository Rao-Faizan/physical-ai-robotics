def plan_grasp_with_cot(object_name: str, scene_description: str) -> Dict:
    """
    ฺู ุขู ุชฺพููน ุฑุฒููฺฏ ฺฉุง ุงุณุชุนูุงู ฺฉุฑุช ูุฆ ฺฏุฑุณ ููุตูุจ ุชุงุฑ ฺฉุฑฺบ
